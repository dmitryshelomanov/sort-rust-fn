## How to use

```rust
    /*
        sort_types
        bubble_sort,
        inerted_sort,
        shake_sort,
        odd_even_sort,
        comb_sort,
    */
    use base_sort::*;
    
    fn main() {
        let mut example = vec![3, 2, 1];

        example.bubble_sort(); // example.[sort_types]()
    }
```
